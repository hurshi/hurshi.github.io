     echo "Shadowsocks turned on"
   elif [[ $1 == "off" ]]; then
     kill -9 $(cat ~/config/sss.pid)
     echo "Shadowsocks turned off"
   else
     echo "Usage: on/off"
   fi''' > ss.sh
